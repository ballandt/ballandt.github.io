# evmath angles between vectors example

from evmath import geo

# Initialise two vectors
a_vec = geo.Vector(2, 4, 9)
b_vec = geo.Vector(4, -9, 0)

# Assign variable to angle
ang = geo.Vector.angle(a_vec, b_vec)

# Get angle in decimal in the needed mode
print(ang.get(mode=geo.Angle.DEG))

# Output 53.21626...
