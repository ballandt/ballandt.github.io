# evmath vector operations example

from evmath import geo

# Initialise two vectors
a_vec = geo.Vector(2, 4, 2)
b_vec = geo.Vector(-3, 2, 9)

# Addition
print(a_vec + b_vec)
# Subtraction
print(a_vec - b_vec)
# Dot product
print(a_vec * b_vec)
# vector product
print(a_vec ** b_vec)
# scalar multiplication
print(2 * a_vec)
